# useHover

通过传入 element 进行 js 模拟 hover 事件

```jsx
const App = () => {
  const element = (hovered: boolean) => (
    <div>Hover me! {hovered && 'Thanks'}</div>
  );

  const [hoverable, hovered] = useHover(element);

  return (
    <div>
      {hoverable}
      <div>{hovered ? 'HOVERED' : ''}</div>
    </div>
  );
};
```
